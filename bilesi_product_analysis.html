generate_occurrences(start_date, recurrence_rule, count=100): """ 根据RRULE生成任务 occurrence recurrence_rule: RRULE格式字符串，如 "FREQ=DAILY;COUNT=30" """ rule = rrule.from_string(recurrence_rule) occurrences = list(rule.between(start_date, datetime.now() + timedelta(days=365), inc=True)) return occurrences[:count] ``` ### 7.2 星星余额计算 ```python def calculate_star_balance(member_id): """ 计算成员当前星星余额 余额 = 累计获得 - 累计消费 - 已撤销兑换 """ transactions = StarTransaction.query.filter_by(member_id=member_id).all() # 排除已撤销的兑换 redemptions = RewardRedemption.query.filter_by( member_id=member_id, status='cancelled' ).all() cancelled_ids = [r.id for r in redemptions] total = 0 for t in transactions: if t.type == 'reward' and t.related_redemption_id in cancelled_ids: continue total += t.amount return total ``` ### 7.3 奖励兑换撤销检查 ```python def can_cancel_redemption(redemption_id): """ 检查兑换是否在可撤销期限内 规则：第二天晚12点前可撤销 """ redemption = RewardRedemption.query.get(redemption_id) if not redemption: return False deadline = redemption.redeemed_at.replace(hour=23, minute=59, second=59) + timedelta(days=1) return datetime.now() < deadline ``` ### 7.4 逾期奖励判定 ```python def check_overdue_completion(task_id, member_id, completed_at): """ 检查任务是否逾期完成 返回：(is_overdue, should_award_stars) """ task = Task.query.get(task_id) # 任务截止时间 task_end = datetime.combine(task.end_date, task.end_time or datetime.min.time()) if completed_at > task_end: return True, task.allow_overdue return False, True ``` --- ## 八、100%复刻检查清单 ### 8.1 功能完整性检查 | 编号 | 功能模块 | 功能点 | 复现状态 | |-----|---------|-------|---------| | 1.1 | 日历 | 周视图 | ⬜
